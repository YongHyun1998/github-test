            class People {
                constructor(name, age, inCountry) { //생성자 함수
                    //속성 설정
                    this.name = name;
                    this.age = age;
                    this.inCountry = inCountry;
                }
 
                //메소드 설정
                printOut() {
                    var inOut = "내국인";
                    if (!this.inCountry) {
                        var inOut = "외국인";
                    }
 
                    return (`안녕하세요~\n저의 이름은 ${this.name}이고 나이는 ${this.age}이며 ${inOut} 입니다~\n잘부탁드립니다!`);
                }
 
                //값을 받는(지정) 방법 //set
                set setAge(age) {
                    if(1 < age && age < 130 && !isNaN(age)) {
                        this.age = age + 1;
                    }
                }
                //값을 리턴하는 방법 //get
                get getAge() {
                    return this.age;
                }
            }
 
            var hong = new People("홍길동", 28, true);
            console.log(hong.printOut());
            // hong.age = 'tes'; //set
            // console.log(hong.age); //get
 
            hong.setAge = 'tett';
            console.log(hong.getAge);
